\section{Script Reference}
\label{sec:AppendixScripts}

This appendix documents all scripts in the \filepath{scripts/} directory.

\subsection{Build and install}

\begin{description}
\item[\texttt{build-archimesh-client.sh}]
  Convenience wrapper that calls \texttt{build-client-standalone.sh}.

\item[\texttt{build-client-standalone.sh}]
  Builds the client plugin using Tycho/Maven with the target platform.

\item[\texttt{build-admin-web.sh}]
  Builds the Svelte admin UI and copies output to the server's static
  resources directory.

\item[\texttt{install-archimesh-plugin.sh}]
  Installs the built client plugin JAR into the local Archi installation's
  dropins directory.

\item[\texttt{sync-archimesh-client.sh}]
  Syncs client plugin source changes without a full rebuild.

\item[\texttt{prepare-client-target-platform.sh}]
  Prepares the Eclipse target platform bundles required for client plugin
  compilation.

\item[\texttt{mint-dev-jwt.sh}]
  Mints a local dev JWT signed with the test private key.\\
  Usage: \texttt{./mint-dev-jwt.sh --user alice --roles admin,model\_writer}\\
  Options: \texttt{--expires-in <seconds>}, \texttt{--audience <aud>},
  \texttt{--print-payload}.\\
  Env overrides: \texttt{DEV\_JWT\_PRIVATE\_KEY}, \texttt{DEV\_JWT\_ISSUER},
  \texttt{DEV\_JWT\_EXPIRES\_IN}, \texttt{DEV\_JWT\_AUDIENCE}.
\end{description}

\subsection{Development infrastructure}

\begin{description}
\item[\texttt{dev-up.sh}]
  Starts Docker Compose infrastructure (Neo4j + Kafka) and optionally the
  server. Usage: \texttt{./dev-up.sh [background]}.
  Supports OIDC mode via \texttt{ARCHIMESH\_IDENTITY\_MODE=oidc}.

\item[\texttt{dev-down.sh}]
  Stops Docker Compose infrastructure.
\end{description}

\subsection{Validation and gate checks}

\begin{description}
\item[\texttt{validate-local.sh}]
  Central validation entry point.\\
  Usage: \texttt{./validate-local.sh <mode>}\\
  Modes:
  \begin{description}
  \item[\texttt{fast}] Lightweight hardening and parity checks (no Docker
    infra needed). Runs \texttt{check-crdt-hardening.sh} and
    \texttt{check-notation-parity.sh}.
  \item[\texttt{gate}] Full client + server CRDT gate via
    \texttt{crdt-gate.sh}.
  \item[\texttt{infra}] Full gate with local Kafka/Neo4j-backed integration
    coverage. Sets \texttt{RUN\_LOCAL\_INFRA\_IT=true}.
  \item[\texttt{ws}] Full gate with websocket end-to-end coverage. Sets
    \texttt{RUN\_WS\_E2E\_IT=true}.
  \end{description}

\item[\texttt{crdt-gate.sh}]
  Runs the CRDT test suite end-to-end. First runs hardening contract checks,
  then installs the client plugin artifact, runs client CRDT tests
  (\texttt{OpMapperNotationInventoryTest}, \texttt{OpMapperConnectionBatchTest},
  \texttt{CrdtEntityMergeTest}, \texttt{CrdtPropertyMergeTest},
  \texttt{CrdtOrSetTest}), then runs server CRDT tests
  (\texttt{ArchimeshServiceTest}, optionally
  \texttt{LocalInfraIntegrationTest} when \texttt{RUN\_LOCAL\_INFRA\_IT=true}).

\item[\texttt{check-crdt-hardening.sh}]
  Preflight contract check that verifies presence of core CRDT test files,
  gate scripts and spec references. Fails fast when test infrastructure
  drifts.

\item[\texttt{check-notation-parity.sh}]
  Compares notation key inventory across: schema definitions
  (\texttt{schemas/ops.json}), server runtime whitelist constants
  (\texttt{ArchimeshService}) and client inventory test expectations
  (\texttt{OpMapperNotationInventoryTest}). Ensures all three agree on
  supported notation fields.

\item[\texttt{check-archimesh-sanity.sh}]
  Scans Archi workspace logs for critical sync issues, integrity warnings
  and connection errors. Useful for diagnosing client-side problems.
\end{description}

\subsection{Log monitoring}

\begin{description}
\item[\texttt{tail-archimesh-logs.sh}]
  Tails Archi workspace logs with archimesh-related filtering. Useful for
  real-time debugging during client plugin development.
\end{description}
